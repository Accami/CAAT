1 /**
  2  * See LICENSE file.
  3  *
  4  * Get realtime Debug information of CAAT's activity.
  5  * Set CAAT.DEBUG=1 before any CAAT.Director object creation.
  6  * This class expects a DOM node called 'caat-debug' being a container element (DIV) where
  7  * it will append itself. If this node is not present, it will append itself to the document's body.
  8  *
  9  */
 10 
 11 (function() {
 12 
 13     CAAT.Debug= function() {
 14         return this;
 15     };
 16 
 17     CAAT.Debug.prototype= {
 18 
 19         width:  0,
 20         height: 0,
 21         canvas: null,
 22         ctx:    null,
 23         statistics: null,
 24 
 25         SCALE:  50,
 26 
 27         setScale : function(s) {
 28             this.scale= s;
 29             return this;
 30         },
 31 
 32         initialize: function(w,h) {
 33             this.width= w;
 34             this.height= h;
 35 
 36             this.canvas= document.createElement('canvas');
 37             this.canvas.width= w;
 38             this.canvas.height=h;
 39             this.ctx= this.canvas.getContext('2d');
 40 
 41             this.ctx.fillStyle= 'black';
 42             this.ctx.fillRect(0,0,this.width,this.height);
 43 
 44             var dom= document.getElementById('caat-debug');
 45             if ( null===dom ) {
 46                 document.body.appendChild( this.canvas );
 47             } else {
 48                 dom.appendChild( this.canvas );
 49             }
 50 
 51             return this;
 52         },
 53 
 54         debugInfo : function( statistics ) {
 55             this.statistics= statistics;
 56             this.paint();
 57         },
 58 
 59         prevRAF: -1,
 60 
 61         paint : function() {
 62             var ctx= this.ctx;
 63             var t=0;
 64 
 65             ctx.drawImage(
 66                 this.canvas,
 67                 1, 0, this.width-1, this.height,
 68                 0, 0, this.width-1, this.height );
 69 
 70             ctx.strokeStyle= 'black';
 71             ctx.beginPath();
 72             ctx.moveTo( this.width-.5, 0 );
 73             ctx.lineTo( this.width-.5, this.height );
 74             ctx.stroke();
 75 
 76             ctx.strokeStyle= 'rgba(0,255,0,.8)';
 77             ctx.beginPath();
 78             t= this.height-((15/this.SCALE*this.height)>>0)-.5;
 79             ctx.moveTo( .5, t );
 80             ctx.lineTo( this.width+.5, t );
 81             ctx.stroke();
 82 
 83             ctx.strokeStyle= 'rgba(255,255,0,.8)';
 84             ctx.beginPath();
 85             t= this.height-((25/this.SCALE*this.height)>>0)-.5;
 86             ctx.moveTo( .5, t );
 87             ctx.lineTo( this.width+.5, t );
 88             ctx.stroke();
 89 
 90             ctx.strokeStyle= CAAT.FRAME_TIME<16 ? 'green' : CAAT.FRAME_TIME<25 ? 'yellow' : 'red';
 91             ctx.beginPath();
 92             ctx.moveTo( this.width-.5, this.height );
 93             ctx.lineTo( this.width-.5, this.height-(CAAT.FRAME_TIME*this.height/this.SCALE) );
 94             ctx.stroke();
 95 
 96             var t1= this.height-(CAAT.REQUEST_ANIMATION_FRAME_TIME/this.SCALE*this.height);
 97             if (-1===this.prevRAF)   {
 98                 this.prevRAF= t1;
 99             }
100 
101             ctx.strokeStyle= 'rgba(255,0,255,.5)';
102             ctx.beginPath();
103             ctx.moveTo( this.width-.5, t1 );
104             ctx.lineTo( this.width-.5, this.prevRAF );
105             ctx.stroke();
106 
107             this.prevRAF= t1;
108 
109             ctx.fillStyle='rgba(255,0,0,.75)';
110             ctx.fillRect( 0,0,180,15);
111             ctx.fillStyle='white';
112             ctx.fillText(
113                     '  Total: '+this.statistics.size_total+
114                     '  Active: '+this.statistics.size_active+
115                     '  Draws: '+this.statistics.draws,
116                     0,
117                     12 );
118         }
119     };
120 })();
