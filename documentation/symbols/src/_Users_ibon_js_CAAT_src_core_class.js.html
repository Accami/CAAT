1 /**
  2  * Extend a prototype with another to form a classical OOP inheritance procedure.
  3  *
  4  * @param subc Prototype to define the base class
  5  * @param superc Prototype to be extended (derived class).
  6  */
  7 function extend(subc, superc) {
  8     var subcp = subc.prototype;
  9 
 10     var F = function() {
 11     };
 12     F.prototype = superc.prototype;
 13 
 14     subc.prototype = new F();       // chain prototypes.
 15     subc.superclass = superc.prototype;
 16     subc.prototype.constructor = subc;
 17 
 18     if (superc.prototype.constructor == Object.prototype.constructor) {
 19         superc.prototype.constructor = superc;
 20     }
 21 
 22     // los metodos de superc, que no esten en esta clase, crear un metodo que
 23     // llama al metodo de superc.
 24     for (var method in subcp) {
 25         if (subcp.hasOwnProperty(method)) {
 26             subc.prototype[method] = subcp[method];
 27 
 28             // tenemos en super un metodo con igual nombre.
 29             if ( superc.prototype[method]) {
 30                 subc.prototype[method]= (function(fn, fnsuper) {
 31                     return function() {
 32                         var prevMethod= this.__super;
 33 
 34                         this.__super= fnsuper;
 35 
 36                         var retValue= fn.apply(
 37                                 this,
 38                                 Array.prototype.slice.call(arguments) );
 39 
 40                         this.__super= prevMethod;
 41 
 42                         return retValue;
 43                     }
 44                 })(subc.prototype[method], superc.prototype[method]);
 45             }
 46         }
 47     }
 48 };

 50 /**
 51  * Proxy an object or wrap/decorate a function.
 52  *
 53  * @param object
 54  * @param preMethod
 55  * @param postMethod
 56  * @param errorMethod
 57  */
 58 function proxy(object, preMethod, postMethod, errorMethod) {
 59 
 60     // proxy a function
 61     if ( typeof object=='function' ) {
 62 
 63         if ( object.__isProxy ) {
 64             return object;
 65         }
 66 
 67         return (function(fn) {
 68             var proxyfn= function() {
 69                 if ( preMethod ) {
 70                     preMethod({
 71                             fn: fn,
 72                             arguments:  Array.prototype.slice.call(arguments)} );
 73                 }
 74                 var retValue= null;
 75                 try {
 76                     // apply original function call with itself as context
 77                     retValue= fn.apply(fn, Array.prototype.slice.call(arguments));
 78                     // everything went right on function call, then call
 79                     // post-method hook if present
 80                     if ( postMethod ) {
 81                         postMethod({
 82                                 fn: fn,
 83                                 arguments:  Array.prototype.slice.call(arguments)} );
 84                     }
 85                 } catch(e) {
 86                     // an exeception was thrown, call exception-method hook if
 87                     // present and return its result as execution result.
 88                     if( errorMethod ) {
 89                         retValue= errorMethod({
 90                             fn: fn,
 91                             arguments:  Array.prototype.slice.call(arguments),
 92                             exception:  e} );
 93                     } else {
 94                         // since there's no error hook, just throw the exception
 95                         throw e;
 96                     }
 97                 }
 98 
 99                 // return original returned value to the caller.
100                 return retValue;
101             }
102             proxyfn.__isProxy= true;
103             return proxyfn;
104 
105         })(object);
106     }
107 
108     /**
109      * If not a function then only non privitive objects can be proxied.
110      * If it is a previously created proxy, return the proxy itself.
111      */
112     if ( !typeof object=='object' ||
113             object.constructor==Array ||
114             object.constructor==String ||
115             object.__isProxy ) {
116 
117         return object;
118     }
119 
120     // Our proxy object class.
121     var cproxy= function() {};
122     // A new proxy instance.
123     var proxy= new cproxy();
124     // hold the proxied object as member. Needed to assign proper
125     // context on proxy method call.
126     proxy.__object= object;
127     proxy.__isProxy= true;
128 
129     // For every element in the object to be proxied
130     for( var method in object ) {
131         // only function members
132         if ( typeof object[method]=='function' ) {
133             // add to the proxy object a method of equal signature to the
134             // method present at the object to be proxied.
135             // cache references of object, function and function name.
136             proxy[method]= (function(proxy,fn,method) {
137                 return function() {
138                     // call pre-method hook if present.
139                     if ( preMethod ) {
140                         preMethod({
141                                 object:     proxy.__object,
142                                 method:     method,
143                                 arguments:  Array.prototype.slice.call(arguments)} );
144                     }
145                     var retValue= null;
146                     try {
147                         // apply original object call with proxied object as
148                         // function context.
149                         retValue= fn.apply( proxy.__object, arguments );
150                         // everything went right on function call, the call
151                         // post-method hook if present
152                         if ( postMethod ) {
153                             postMethod({
154                                     object:     proxy.__object,
155                                     method:     method,
156                                     arguments:  Array.prototype.slice.call(arguments)} );
157                         }
158                     } catch(e) {
159                         // an exeception was thrown, call exception-method hook if
160                         // present and return its result as execution result.
161                         if( errorMethod ) {
162                             retValue= errorMethod({
163                                 object:     proxy.__object,
164                                 method:     method,
165                                 arguments:  Array.prototype.slice.call(arguments),
166                                 exception:  e} );
167                         } else {
168                             // since there's no error hook, just throw the exception
169                             throw e;
170                         }
171                     }
172 
173                     // return original returned value to the caller.
174                     return retValue;
175                 }
176             })(proxy,object[method],method);
177         }
178     }
179 
180     // return our newly created and populated of functions proxy object.
181     return proxy;
182 };
