  3  *
  4  * Sound implementation.
  5  */
  6 
  7 (function() {
  8 
  9     /**
 10      * This class is a sound manager implementation which can play at least 'numChannels' sounds at the same time.
 11      * By default, CAAT.Director instances will set eight channels to play sound.
 12      * <p>
 13      * If more than 'numChannels' sounds want to be played at the same time the requests will be dropped,
 14      * so no more than 'numChannels' sounds can be concurrently played.
 15      * <p>
 16      * Available sounds to be played must be supplied to every CAAT.Director instance by calling <code>addSound</code>
 17      * method. The default implementation will accept a URL/URI or a HTMLAudioElement as source.
 18      * <p>
 19      * The cached elements can be played, or looped. The <code>loop</code> method will return a handler to
 20      * give the opportunity of cancelling the sound.
 21      * <p>
 22      * Be aware of Audio.canPlay, is able to return 'yes', 'no', 'maybe', ..., so anything different from
 23      * '' and 'no' will do.
 24      *
 25      * @constructor
 26      *
 27      */
 28     CAAT.AudioManager= function() {
 29         return this;
 30     };
 31 
 32     CAAT.AudioManager.prototype= {
 33 
 34         musicEnabled:       true,
 35         fxEnabled:          true,
 36         audioCache:         null,   // audio elements.
 37         channels:           null,   // available playing channels.
 38         workingChannels:    null,   // currently playing channels.
 39         loopingChannels:    [],
 40         audioTypes: {               // supported audio formats. Don't remember where i took them from :S
 41 	        'mp3': 'audio/mpeg;',
 42             'ogg': 'audio/ogg; codecs="vorbis"',
 43             'wav': 'audio/wav; codecs="1"',
 44             'mp4': 'audio/mp4; codecs="mp4a.40.2"'
 45 		},
 46 
 47         /**
 48          * Initializes the sound subsystem by creating a fixed number of Audio channels.
 49          * Every channel registers a handler for sound playing finalization. If a callback is set, the
 50          * callback function will be called with the associated sound id in the cache.
 51          *
 52          * @param numChannels {number} number of channels to pre-create. 8 by default.
 53          *
 54          * @return this.
 55          */
 56         initialize : function(numChannels) {
 57 
 58             this.audioCache=      [];
 59             this.channels=        [];
 60             this.workingChannels= [];
 61 
 62             for( var i=0; i<numChannels; i++ ) {
 63                 var channel= document.createElement('audio');
 64 
 65                 if ( null!=channel ) {
 66                     channel.finished= -1;
 67                     this.channels.push( channel );
 68                     var me= this;
 69                     channel.addEventListener(
 70                             'ended',
 71                             // on sound end, set channel to available channels list.
 72                             function(audioEvent) {
 73                                 var target= audioEvent.target;
 74                                 var i;
 75 
 76                                 // remove from workingChannels
 77                                 for( i=0; i<me.workingChannels.length; i++ ) {
 78                                     if (me.workingChannels[i]==target ) {
 79                                         me.workingChannels.splice(i,1);
 80                                         break;
 81                                     }
 82                                 }
 83 
 84                                 if ( target.caat_callback ) {
 85                                     target.caat_callback(target.caat_id);
 86                                 }
 87 
 88                                 // set back to channels.
 89                                 me.channels.push(target);
 90                             },
 91                             false
 92                     );
 93                 }
 94             }
 95 
 96             return this;
 97         },
 98         /**
 99          * creates an Audio object and adds it to the audio cache in case the url points to a
100          * suitable audio file to be played.
101          *
102          * @param id {object} an object to reference the audio object. Tipically a string.
103          * @param url {string|HTMLElement} an url pointing to an audio resource or an HTMLAudioElement
104          * object.
105          * @param endplaying_callback {function} a callback function to notify on audio finalization. The
106          * function is of the form <code>function(id{string})</code>. The id parameter is the associated id
107          * in the cache.
108          *
109          * @return this
110          */
111         addAudio : function( id, url, endplaying_callback ) {
112             var audio= null;
113             var extension= null;
114 
115             if ( typeof url == "string" ) {
116 
117                 audio= document.createElement('audio');
118                 if ( null!=audio ) {
119 
120                     if(!audio.canPlayType) {
121                         return this;
122                     }
123 
124                     extension= url.substr(url.lastIndexOf('.')+1);
125                     var canplay= audio.canPlayType(this.audioTypes[extension]);
126 
127                     if(canplay!=="" && canplay!=="no") {
128                         audio.src= url;
129                         audio.preload = "auto";
130                         audio.load();
131                         if ( endplaying_callback ) {
132                             audio.caat_callback= endplaying_callback;
133                             audio.caat_id= id;
134                         }
135                         this.audioCache.push( { id:id, audio:audio } );
136                     }
137                 }
138             } else {
139                 try {
140                     if ( url instanceof HTMLAudioElement ) {
141                         audio= url;
142                         extension= audio.src.substr(audio.src.lastIndexOf('.')+1);
143                         if ( audio.canPlayType(this.audioTypes[extension]) ) {
144                             if ( endplaying_callback ) {
145                                 audio.caat_callback= endplaying_callback;
146                                 audio.caat_id= id;
147                             }
148                             this.audioCache.push( { id:id, audio:audio } );
149                         }
150                     }
151                 }
152                 catch(e) {
153                     
154                 }
155             }
156 
157 
158             return this;
159         },
160         /**
161          * Returns an audio object.
162          * @param aId {object} the id associated to the target Audio object.
163          * @return {object} the HTMLAudioElement addociated to the given id.
164          */
165         getAudio : function(aId) {
166             for( var i=0; i<this.audioCache.length; i++ ) {
167                 if ( this.audioCache[i].id==aId ) {
168                     return this.audioCache[i].audio;
169                 }
170             }
171 
172             return null;
173         },
174         /**
175          * Plays an audio file from the cache if any sound channel is available.
176          * The playing sound will occupy a sound channel and when ends playing will leave
177          * the channel free for any other sound to be played in.
178          * @param id {object} an object identifying a sound in the sound cache.
179          * @return this.
180          */
181         play : function( id ) {
182             if ( !this.fxEnabled ) {
183                 return this;
184             }
185 
186             var audio= this.getAudio(id);
187             // existe el audio, y ademas hay un canal de audio disponible.
188             if ( null!=audio && this.channels.length>0 ) {
189                 var channel= this.channels.shift();
190                 channel.src= audio.src;
191                 channel.load();
192                 channel.play();
193                 this.workingChannels.push(channel);
194             }
195 
196             return this;
197         },
198         /**
199          * This method creates a new AudioChannel to loop the sound with.
200          * It returns an Audio object so that the developer can cancel the sound loop at will.
201          * The user must call <code>pause()</code> method to stop playing a loop.
202          *
203          * @return {HTMLElement|null} an Audio instance if a valid sound id is supplied. Null otherwise
204          */
205         loop : function( id ) {
206 
207             if (!this.musicEnabled) {
208                 return this;
209             }
210 
211             var audio_in_cache= this.getAudio(id);
212             // existe el audio, y ademas hay un canal de audio disponible.
213             if ( null!=audio_in_cache ) {
214                 var audio= document.createElement('audio');
215                 if ( null!=audio ) {
216                     audio.src= audio_in_cache.src;
217                     audio.preload = "auto";
218                     audio.loop= true;
219                     audio.load();
220                     audio.play();
221                     this.loopingChannels.push(audio);
222                     return audio;
223                 }
224             }
225 
226             return null;
227         },
228         /**
229          * Cancel all playing audio channels
230          * Get back the playing channels to available channel list.
231          *
232          * @return this
233          */
234         endSound : function() {
235             for( var i=0; i<this.workingChannels.length; i++ ) {
236                 this.workingChannels[i].pause();
237                 this.channels.push( this.workingChannels[i] );
238             }
239 
240             return this;
241         },
242         setSoundEffectsEnabled : function( enable ) {
243             this.fxEnabled= enable;
244             return this;
245         },
246         isSoundEffectsEnabled : function() {
247             return this.fxEnabled;
248         },
249         setMusicEnabled : function( enable ) {
250             this.musicEnabled= enable;
251             for( var i=0; i<this.loopingChannels.length; i++ ) {
252                 if ( enable ) {
253                     this.loopingChannels[i].play();
254                 } else {
255                     this.loopingChannels[i].pause();
256                 }
257             }
258             return this;
259         },
260         isMusicEnabled : function() {
261             return this.musicEnabled;
262         }
263     };
264 })();
