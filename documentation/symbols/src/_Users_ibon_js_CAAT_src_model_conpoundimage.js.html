1 /**

  3  *
  4  * ConpoundBitmap handles subimaging from an image.
  5  * It's useful for SpriteActor class.
  6  *

 11 
 12 (function() {
 13 
 14     /**
 15      * A CompoundImage is an sprite sheet. It encapsulates an Image and treates and references it as a two
 16      * dimensional array of row by columns sub-images. The access form will be sequential so if defined a
 17      * CompoundImage of more than one row, the subimages will be referenced by an index ranging from 0 to
 18      * rows*columns-1. Each sumimage will be of size (image.width/columns) by (image.height/rows).
 19      *
 20      * <p>
 21      * It is able to draw its sub-images in the following ways:
 22      * <ul>
 23      * <li>no transformed (default)
 24      * <li>flipped horizontally
 25      * <li>flipped vertically
 26      * <li>flipped both vertical and horizontally
 27      * </ul>
 28      *
 29      * <p>
 30      * It is supposed to be used in conjunction with <code>CAAT.SpriteActor</code> instances.
 31      *
 32      * @constructor
 33      *
 34      */
 35     CAAT.CompoundImage = function() {
 36         this.paint= this.paintN;
 37         return this;
 38     };
 39 
 40     CAAT.CompoundImage.prototype = {
 41 
 42         NORMAL:                     1,
 43         INV_VERTICAL:                 2,
 44         INV_HORIZONTAL:             4,
 45         INV_VERTICAL_AND_HORIZONTAL:8,
 46 
 47         image:                        null,
 48         rows:                        0,
 49         cols:                        0,
 50         width:                        0,
 51         height:                        0,
 52         singleWidth:                0,
 53         singleHeight:                0,
 54 
 55         xyCache:                    null,
 56 
 57         /**
 58          * Initialize a grid of subimages out of a given image.
 59          * @param image {HTMLImageElement|Image} an image object.
 60          * @param rows {number} number of rows.
 61          * @param cols {number} number of columns
 62          *
 63          * @return this
 64          */
 65         initialize : function(image, rows, cols) {
 66             this.image = image;
 67             this.rows = rows;
 68             this.cols = cols;
 69             this.width = image.width;
 70             this.height = image.height;
 71             this.singleWidth = Math.floor(this.width / cols);
 72             this.singleHeight = Math.floor(this.height / rows);
 73             this.xyCache = [];
 74 
 75             var i,sx0,sy0;
 76             if (image.__texturePage) {
 77                 image.__du = this.singleWidth / image.__texturePage.width;
 78                 image.__dv = this.singleHeight / image.__texturePage.height;
 79 
 80 
 81                 var w = this.singleWidth;
 82                 var h = this.singleHeight;
 83                 var mod = this.cols;
 84                 if (image.inverted) {
 85                     var t = w;
 86                     w = h;
 87                     h = t;
 88                     mod = this.rows;
 89                 }
 90 
 91                 var xt = this.image.__tx;
 92                 var yt = this.image.__ty;
 93 
 94                 var tp = this.image.__texturePage;
 95 
 96                 for (i = 0; i < rows * cols; i++) {
 97 
 98 
 99                     var c = ((i % mod) >> 0);
100                     var r = ((i / mod) >> 0);
101 
102                     var u = xt + c * w;  // esquina izq x
103                     var v = yt + r * h;
104 
105                     var u1 = u + w;
106                     var v1 = v + h;
107 
108                     /*
109                      var du= image.__du;
110                      var dv= image.__dv;
111                      var mod= this.cols;
112                      if ( image.inverted) {
113                      var t= du;
114                      du= dv;
115                      dv= t;
116                      mod= this.rows;
117                      }
118 
119                      sx0= ((i%mod)>>0)*du;
120                      sy0= ((i/mod)>>0)*dv;
121 
122                      var u= image.__u+sx0;
123                      var v= image.__v+sy0;
124 
125                      var u1= u+du;
126                      var v1= v+dv;
127                      */
128 
129                     this.xyCache.push([u / tp.width,v / tp.height,u1 / tp.width,v1 / tp.height,u,v,u1,v1]);
130                 }
131 
132             } else {
133                 for (i = 0; i < rows * cols; i++) {
134                     sx0 = ((i % this.cols) | 0) * this.singleWidth;
135                     sy0 = ((i / this.cols) | 0) * this.singleHeight;
136 
137                     this.xyCache.push([sx0,sy0]);
138                 }
139             }
140 
141             return this;
142         },
143         /**
144          * Draws the subimage pointed by imageIndex horizontally inverted.
145          * @param canvas a canvas context.
146          * @param imageIndex {number} a subimage index.
147          * @param x {number} x position in canvas to draw the image.
148          * @param y {number} y position in canvas to draw the image.
149          *
150          * @return this
151          */
152         paintInvertedH : function(canvas, imageIndex, x, y) {
153 
154             canvas.save();
155             canvas.translate(((0.5 + x) | 0) + this.singleWidth, (0.5 + y) | 0);
156             canvas.scale(-1, 1);
157 
158             canvas.drawImage(this.image,
159                     this.xyCache[imageIndex][0], this.xyCache[imageIndex][1],
160                     this.singleWidth, this.singleHeight,
161                     0, 0, this.singleWidth, this.singleHeight);
162 
163             canvas.restore();
164 
165             return this;
166         },
167         /**
168          * Draws the subimage pointed by imageIndex vertically inverted.
169          * @param canvas a canvas context.
170          * @param imageIndex {number} a subimage index.
171          * @param x {number} x position in canvas to draw the image.
172          * @param y {number} y position in canvas to draw the image.
173          *
174          * @return this
175          */
176         paintInvertedV : function(canvas, imageIndex, x, y) {
177 
178             canvas.save();
179             canvas.translate((x + 0.5) | 0, (0.5 + y + this.singleHeight) | 0);
180             canvas.scale(1, -1);
181 
182             canvas.drawImage(
183                     this.image,
184                     this.xyCache[imageIndex][0], this.xyCache[imageIndex][1],
185                     this.singleWidth, this.singleHeight,
186                     0, 0, this.singleWidth, this.singleHeight);
187 
188             canvas.restore();
189 
190             return this;
191         },
192         /**
193          * Draws the subimage pointed by imageIndex both horizontal and vertically inverted.
194          * @param canvas a canvas context.
195          * @param imageIndex {number} a subimage index.
196          * @param x {number} x position in canvas to draw the image.
197          * @param y {number} y position in canvas to draw the image.
198          *
199          * @return this
200          */
201         paintInvertedHV : function(canvas, imageIndex, x, y) {
202 
203             canvas.save();
204             canvas.translate((x + 0.5) | 0, (0.5 + y + this.singleHeight) | 0);
205             canvas.scale(1, -1);
206             canvas.translate(this.singleWidth, 0);
207             canvas.scale(-1, 1);
208 
209             canvas.drawImage(
210                     this.image,
211                     this.xyCache[imageIndex][0], this.xyCache[imageIndex][1],
212                     this.singleWidth, this.singleHeight,
213                     0, 0, this.singleWidth, this.singleHeight);
214 
215             canvas.restore();
216 
217             return this;
218         },
219         /**
220          * Draws the subimage pointed by imageIndex.
221          * @param canvas a canvas context.
222          * @param imageIndex {number} a subimage index.
223          * @param x {number} x position in canvas to draw the image.
224          * @param y {number} y position in canvas to draw the image.
225          *
226          * @return this
227          */
228         paintN : function(canvas, imageIndex, x, y) {
229             canvas.drawImage(
230                     this.image,
231                     this.xyCache[imageIndex][0]>>0, this.xyCache[imageIndex][1]>>0,
232                     this.singleWidth, this.singleHeight,
233                     x>>0, y>>0, this.singleWidth, this.singleHeight);
234 
235             return this;
236         },
237         paint : function(canvas, imageIndex, x, y) {
238             return this.paintN(canvas,imageIndex,x,y);
239         },
240         /**
241          * Draws the subimage pointed by imageIndex scaled to the size of w and h.
242          * @param canvas a canvas context.
243          * @param imageIndex {number} a subimage index.
244          * @param x {number} x position in canvas to draw the image.
245          * @param y {number} y position in canvas to draw the image.
246          * @param w {number} new width of the subimage.
247          * @param h {number} new height of the subimage.
248          *
249          * @return this
250          */
251         paintScaled : function(canvas, imageIndex, x, y, w, h) {
252             canvas.drawImage(
253                     this.image,
254                     this.xyCache[imageIndex][0], this.xyCache[imageIndex][1],
255                     this.singleWidth, this.singleHeight,
256                     (x + 0.5) | 0, (y + 0.5) | 0, w, h);
257 
258             return this;
259         },
260         /**
261          * Get the number of subimages in this compoundImage
262          * @return {number}
263          */
264         getNumImages : function() {
265             return this.rows * this.cols;
266         },
267         setUV : function(imageIndex, uvBuffer, uvIndex) {
268             var im = this.image;
269 
270             if (!im.__texturePage) {
271                 return;
272             }
273 
274             var index = uvIndex;
275 
276             if (im.inverted) {
277                 uvBuffer[index++] = this.xyCache[imageIndex][2];
278                 uvBuffer[index++] = this.xyCache[imageIndex][1];
279 
280                 uvBuffer[index++] = this.xyCache[imageIndex][2];
281                 uvBuffer[index++] = this.xyCache[imageIndex][3];
282 
283                 uvBuffer[index++] = this.xyCache[imageIndex][0];
284                 uvBuffer[index++] = this.xyCache[imageIndex][3];
285 
286                 uvBuffer[index++] = this.xyCache[imageIndex][0];
287                 uvBuffer[index++] = this.xyCache[imageIndex][1];
288             } else {
289                 uvBuffer[index++] = this.xyCache[imageIndex][0];
290                 uvBuffer[index++] = this.xyCache[imageIndex][1];
291 
292                 uvBuffer[index++] = this.xyCache[imageIndex][2];
293                 uvBuffer[index++] = this.xyCache[imageIndex][1];
294 
295                 uvBuffer[index++] = this.xyCache[imageIndex][2];
296                 uvBuffer[index++] = this.xyCache[imageIndex][3];
297 
298                 uvBuffer[index++] = this.xyCache[imageIndex][0];
299                 uvBuffer[index++] = this.xyCache[imageIndex][3];
300             }
301 
302             //director.uvIndex= index;
303         }
304     };
305 })();
