1 /**
  2  * @author  Hyperandroid  ||  http://hyperandroid.com/
  3  *
  4  * MouseEvent is a class to hold necessary information of every mouse event related to concrete
  5  * scene graph Actors.
  6  *
  7  * Here it is also the logic to on mouse events, pump the correct event to the appropiate scene
  8  * graph Actor.
  9  *
 10  * 20101008 Hyperandroid. changed event scope from CAAT.director.canvas to window. Works under
 11  *          al major browsers on linux and win7. Thanks @alteredq for this tip.
 12  *
 13  * TODO: add events for event pumping:
 14  *  + cancelBubling
 15  *
 16  **/
 17 (function() {
 18 	CAAT.MouseEvent = function() {
 19 		this.point= new CAAT.Point();
 20 		this.screenPoint= new CAAT.Point();
 21 		return this;
 22 	};
 23 	
 24 	CAAT.MouseEvent.prototype= {
 25 		screenPoint:	null,
 26 		point:			null,
 27 		modifiers:		0,
 28 		time:			0,
 29 		source:			null,
 30 		
 31 		SHIFT:			16,
 32 		CONTROL:		17,
 33 		ALT:			18,
 34 
 35 		SHIFT_MASK:		1,
 36 		CONTROL_MASK:	2,
 37 		ALT_MASK:		4,
 38 		
 39 		init : function( x,y,modifiers,source,screenPoint ) {
 40 			this.point.set(x,y);
 41 			this.modifiers= modifiers;
 42 			this.source= source;
 43 			this.screenPoint= screenPoint;
 44 			return this;
 45 		},
 46 		isAltDown : function() {
 47 			return this.modifiers&this.ALT_MASK;
 48 		},
 49 		isControlDown : function() {
 50 			return this.modifiers&this.CONTROL_MASK;
 51 		},
 52 		isShiftDown : function() {
 53 			return this.modifiers&this.SHIFT_MASK;
 54 		}
 55 	};
 56 })();

107 /**
108  * Enable at window level accelerometer events.
109  */
110 (function() {
111 
112     //----------- Test for Accelerometer enabled functions.
113     try {
114         if (window.DeviceMotionEvent != undefined) {
115             CAAT.prevOnDeviceMotion= window.ondevicemotion;
116             window.ondevicemotion = CAAT.onDeviceMotion= function(e) {
117                 CAAT.accelerationIncludingGravity= {
118                     x: e.accelerationIncludingGravity.x,
119                     y: e.accelerationIncludingGravity.y,
120                     z: e.accelerationIncludingGravity.z
121                 };
122 
123                 if ( e.rotationRate ) {
124                     CAAT.rotationRate= {
125                         alpha : e.rotationRate.alpha,
126                         beta  : e.rotationRate.beta,
127                         gamma : e.rotationRate.gamma
128                     };
129                 }
130             }
131         }
132     } catch (e) {
133         // eat it.
134     }
135 
136 })();
