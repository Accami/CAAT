107 CAAT.getCanvasCoord= function __getCanvasCoord(point, e) {
108 	var posx = 0;
109 	var posy = 0;
110 	if (!e) e = window.event;
111 
112     if (e.pageX || e.pageY) 	{
113 		posx = e.pageX;
114 		posy = e.pageY;
115 	}
116 
117 	else if (e.clientX || e.clientY) 	{
118 		posx = e.clientX + document.body.scrollLeft + document.documentElement.scrollLeft;
119 		posy = e.clientY + document.body.scrollTop  + document.documentElement.scrollTop;
120 	}
121 
122     var pposx;
123     var pposy;
124     var i;
125 
126     for( i=0; i<CAAT.director.length; i++ ) {
127         pposx= posx;
128         pposy= posy;
129 
130         var node= CAAT.director[i].canvas;
131 
132         /**
133          * traverse throught the hierarchy of contained html elements to get the correct coordinates offset
134          * on screen.
135          */
136         var px= -1, py=-1;
137 
138         while( false==node instanceof HTMLBodyElement ) {
139 
140             if ( px!=node.offsetLeft && py!=node.offsetTop ) {
141     	        pposx-= node.offsetLeft;
142     	        pposy-= node.offsetTop;
143                 px= node.offsetLeft;
144                 py= node.offsetTop;
145             }
146             node= node.parentNode ? node.parentNode : node.parentElement;
147         }
148 
149 //        console.log('enter ('+posx+','+posy+')  related ('+pposx+','+pposy+')');
150 
151         if ( CAAT.director[i].contains(pposx, pposy) ) {
152             CAAT.targetDirector= CAAT.director[i];
153             point.set(pposx,pposy);
154             CAAT.screenMousePoint.set(pposx, pposy);
155             return;
156         }
157     }
158 
159     CAAT.targetDirector=null;
160 };

240             switch( key ) {
241             case CAAT.MouseEvent.prototype.SHIFT:
242                 CAAT.modifiers&=~CAAT.MouseEvent.prototype.SHIFT_MASK;

244             case CAAT.MouseEvent.prototype.CONTROL:
245                 CAAT.modifiers&=~CAAT.MouseEvent.prototype.CONTROL_MASK;

250             case 68:    // D
251                 if ( CAAT.DEBUG ) {
252                     if ( null!=CAAT.targetDirector ) {
253                         CAAT.targetDirector.debug= !CAAT.targetDirector.debug;
254                     }
255                 }

291     window.addEventListener('mousedown',
292             CAAT.mouseDownFunc= function(e) {
293 
294                 CAAT.getCanvasCoord(CAAT.mousePoint, e);
295 
296                 if ( null==CAAT.targetDirector ) {
297                     return;
298                 }
299                 CAAT.mouseDown = true;
300                 CAAT.lastSelectedActor = CAAT.targetDirector.findActorAtPosition(CAAT.mousePoint);
301                 var px= CAAT.mousePoint.x;
302                 var py= CAAT.mousePoint.y;
303                 
304                 if (null != CAAT.lastSelectedActor) {
305                     // to calculate mouse drag threshold
306                     CAAT.prevMousePoint.x= px;
307                     CAAT.prevMousePoint.y= py;
308                     CAAT.lastSelectedActor.mouseDown(
309                             new CAAT.MouseEvent().init(
310                                     CAAT.lastSelectedActor.rpoint.x,
311                                     CAAT.lastSelectedActor.rpoint.y,
312                                     CAAT.modifiers,
313                                     CAAT.lastSelectedActor,
314                                     CAAT.screenMousePoint));
315                 }
316             },
317             false);

342                 if (null != CAAT.lastSelectedActor) {
343                     CAAT.lastSelectedActor.mouseExit(new CAAT.MouseEvent().init(0, 0, CAAT.modifiers, CAAT.lastSelectedActor, CAAT.screenMousePoint));
344                     CAAT.lastSelectedActor = null;
345                 }
346                 CAAT.mouseDown = false;
347             },

350     window.addEventListener('mousemove',
351             CAAT.mouseMoveFunc= function(e) {
352 
353                 CAAT.getCanvasCoord(CAAT.mousePoint, e);
354                 if ( null==CAAT.targetDirector ) {
355                     return;
356                 }
357 
358                 // drag
359                 if (CAAT.mouseDown && null != CAAT.lastSelectedActor) {
360 
361                     // check for mouse move threshold.
362                     if ( !CAAT.dragging ) {
363                         if ( Math.abs(CAAT.prevMousePoint.x-CAAT.mousePoint.x)< CAAT.DRAG_THRESHOLD_X &&
364                              Math.abs(CAAT.prevMousePoint.y-CAAT.mousePoint.y)< CAAT.DRAG_THRESHOLD_Y ) {
365                             return;
366                         }
367                     }
368 
369                     CAAT.dragging= true;
370                     if (null != CAAT.lastSelectedActor.parent) {
371                         CAAT.lastSelectedActor.parent.inverseTransformCoord(CAAT.mousePoint);
372                     }
373                     CAAT.lastSelectedActor.mouseDrag(
374                             new CAAT.MouseEvent().init(
375                                     CAAT.mousePoint.x,
376                                     CAAT.mousePoint.y,
377                                     CAAT.modifiers,
378                                     CAAT.lastSelectedActor,
379                                     CAAT.screenMousePoint));
380                     return;
381                 }
382 
383                 var lactor = CAAT.targetDirector.findActorAtPosition(CAAT.mousePoint);
384 
385                 // cambiamos de actor.
386                 if (lactor != CAAT.lastSelectedActor) {
387                     if (null != CAAT.lastSelectedActor) {
388                         CAAT.lastSelectedActor.mouseExit(
389                                 new CAAT.MouseEvent().init(
390                                         CAAT.mousePoint.x,
391                                         CAAT.mousePoint.y,
392                                         CAAT.modifiers,
393                                         CAAT.lastSelectedActor,
394                                         CAAT.screenMousePoint));
395                     }
396                     if (null != lactor) {
397                         lactor.mouseEnter(
398                                 new CAAT.MouseEvent().init(
399                                         lactor.rpoint.x,
400                                         lactor.rpoint.y,
401                                         CAAT.modifiers,
402                                         lactor,
403                                         CAAT.screenMousePoint));
404                     }
405                 }
406                 CAAT.lastSelectedActor = lactor;
407                 if (null != lactor) {
408                     CAAT.lastSelectedActor.mouseMove(
409                             new CAAT.MouseEvent().init(
410                                     CAAT.lastSelectedActor.rpoint.x,
411                                     CAAT.lastSelectedActor.rpoint.y,
412                                     CAAT.modifiers,
413                                     CAAT.lastSelectedActor,
414                                     CAAT.screenMousePoint));
415                 }
416             },
417             false);

434     function touchHandler(event) {
435         var touches = event.changedTouches,
436             first = touches[0],
437             type = "";
438 
439         switch (event.type) {
440             case "touchstart": type = "mousedown"; break;
441             case "touchmove":  type = "mousemove"; break;
442             case "touchend":   type = "mouseup"; break;
443             default: return;
444         }
445 
446         //initMouseEvent(type, canBubble, cancelable, view, clickCount,
447         //           screenX, screenY, clientX, clientY, ctrlKey,
448         //           altKey, shiftKey, metaKey, button, relatedTarget);
449 
450         var simulatedEvent = document.createEvent("MouseEvent");
451         simulatedEvent.initMouseEvent(
452                 type,
453                 true,
454                 true,
455                 window,
456                 1,
457                 first.screenX,
458                 first.screenY,
459                 first.clientX,
460                 first.clientY,
461                 false,
462                 false,
463                 false,
464                 false,
465                 0/*left*/,
466                 null);
467 
468         window.dispatchEvent(simulatedEvent);
469         //first.target.dispatchEvent(simulatedEvent);
470         event.preventDefault();
471     }
472 
473     window.addEventListener("touchstart", touchHandler, true);
474     window.addEventListener("touchmove", touchHandler, true);
475     window.addEventListener("touchend", touchHandler, true);
476     window.addEventListener("touchcancel", touchHandler, true);
