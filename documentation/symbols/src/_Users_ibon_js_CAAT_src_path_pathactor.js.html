1 /**
  2  * @author  Hyperandroid  ||  http://hyperandroid.com/
  3  *
  4  * An actor to show the path and its handles in the scene graph. 
  5  *
  6  **/
  7 (function() {
  8     /**
  9      * This class paints and handles the interactive behavior of a path.
 10      *
 11      * @constructor
 12      * @extends CAAT.ActorContainer
 13      */
 14 	CAAT.PathActor= function() {
 15 		CAAT.PathActor.superclass.constructor.call(this);
 16 		return this;
 17 	};
 18 	
 19 	CAAT.PathActor.prototype= {
 20 		path:					null,
 21 		pathBoundingRectangle:	null,
 22 		bOutline:				false,
 23 
 24         /**
 25          * Return the contained path.
 26          * @return {CAAT.Path}
 27          */
 28         getPath : function() {
 29             return this.path;
 30         },
 31         /**
 32          * Sets the path to manage.
 33          * @param path {CAAT.PathSegment}
 34          * @return this
 35          */
 36 		setPath : function(path) {
 37 			this.path= path;
 38 			this.pathBoundingRectangle= path.getBoundingBox();
 39             return this;
 40 		},
 41         /**
 42          * Paint this actor.
 43          * @param director {CAAT.Director}
 44          * @param time {number}. Scene time.
 45          */
 46 		paint : function(director, time) {
 47 
 48             var canvas= director.crc;
 49 
 50             canvas.strokeStyle='black';
 51 			this.path.paint(director);
 52 			
 53 			if ( this.bOutline ) {
 54 				canvas.strokeStyle='black';
 55 				canvas.strokeRect(0,0,this.width,this.height);
 56 			}
 57 		},
 58         /**
 59          * Route mouse dragging functionality to the contained path.
 60          * @param mouseEvent {CAAT.MouseEvent}
 61          */
 62 		mouseDrag : function(mouseEvent) {
 63 			this.path.drag(mouseEvent.point.x, mouseEvent.point.y);
 64 		},
 65         /**
 66          * Route mouse down functionality to the contained path.
 67          * @param mouseEvent {CAAT.MouseEvent}
 68          */
 69 		mouseDown : function(mouseEvent) {
 70 			this.path.press(mouseEvent.point.x, mouseEvent.point.y);
 71 		},
 72         /**
 73          * Route mouse up functionality to the contained path.
 74          * @param mouseEvent {CAAT.MouseEvent}
 75          */
 76 		mouseUp : function(mouseEvent) {
 77 			this.path.release();
 78 		}
 79 	};
 80 
 81     extend( CAAT.PathActor, CAAT.ActorContainer, null);
 82 })();
